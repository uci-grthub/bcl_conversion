At the end do renaming and make md5sum, size, and count files these files used for making notifications

rename L8 L8-G1 xR074-L8-P*

sed 's/L8/L8-G1/' md5sum_lane8.txt > md5sum_lane8-group1.txt

grep Read xR074-L8-G1-*.txt | cut -d : -f3 | sed 's/ //' > count-lane8-group1.txt

ls xR074-L8-G1-*.png > image-list-lane8-group1.txt
